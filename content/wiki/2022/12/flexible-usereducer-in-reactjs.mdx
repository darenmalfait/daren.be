This contains a snippet for a flexible way of using useReducer. I tend to use this if my component has a lot of states. It's cleaner than `useState`.

```javascript
import * as React from 'react'

const initialState = {
  key: "value",
};

function Component() {
  const reducer = (state, newState) => { return { ...state, ...newState } };
  const [store, setStore] = React.useReducer(reducer, initialState);

  function updateStore(values) {
    setStore({ ...values });
  }

  return <div />
}
```
